2️⃣ COMPLETE PRISMA SCHEMA (ENTERPRISE READY)

Here is the consolidated production structure:

model User {
  id            String   @id @default(uuid())
  name          String
  email         String   @unique
  password      String
  role          Role     @default(MARKETER)
  subscription  SubscriptionTier
  createdAt     DateTime @default(now())
  inventories   Inventory[]
}

model Terminal {
  id        String   @id @default(uuid())
  name      String
  state     String
  active    Boolean  @default(true)
  depots    Depot[]
  forecasts Forecast[]
}

model Forecast {
  id              String   @id @default(uuid())
  terminalId      String
  productType     Product
  bias            MarketBias
  probability     Float
  expectedMin     Float
  expectedMax     Float
  confidence      Float
  createdAt       DateTime @default(now())
}

model Depot {
  id          String @id @default(uuid())
  name        String
  terminalId  String
  prices      DepotPrice[]
}

model DepotPrice {
  id          String   @id @default(uuid())
  depotId     String
  productType Product
  price       Float
  updatedAt   DateTime @default(now())
}

model Inventory {
  id           String   @id @default(uuid())
  userId       String
  terminalId   String
  productType  Product
  volume       Float
  averageCost  Float
}

enum Product {
  PMS
  AGO
  JET_A1
  ATK
  LPG
}

enum MarketBias {
  Bullish
  Bearish
  Neutral
}

enum Role {
  ADMIN
  MARKETER
}

enum SubscriptionTier {
  FREE
  PRO
  ELITE
}
